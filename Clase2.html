vec.esc<-function (vector, escalar=1){
  
  potencia<-function(escalar){
    escalar^2
  }
  
  potencia(escalar)*vector
}

vec.esc(c(1,2,1))
## [1] 1 2 1
vec.esc(c(1,1,1),2)
## [1] 4 4 4
